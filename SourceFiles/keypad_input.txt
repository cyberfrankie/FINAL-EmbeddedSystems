/*
    ### USED ON THE ARDUINO ONLY ###
*/

#include <Keypad.h>

const byte ROWS = 4, COLS = 4;
char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
byte rowPins[ROWS] = {9, 8, 7, 6};
byte colPins[COLS] = {5, 4, 3, 2};

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

const String CORRECT_CODE = "369B";
String inputCode = "";

void loop() {
  char key = keypad.getKey();
  if (key) {
    if (key == '*') {
      inputCode = "";
      Serial.println("CLEARED");
    } else if (key == '#') {
      if (inputCode == "") {
        Serial.println("REARM");
      } else if (inputCode == CORRECT_CODE) {
        Serial.println("CORRECT");
      } else {
        Serial.println("INCORRECT");
      }
      inputCode = "";
    } else if (key == 'D') {
      Serial.println("DOORBELL");
    } else {
      inputCode += key;
      Serial.println("KEY");
    }
  }
}
